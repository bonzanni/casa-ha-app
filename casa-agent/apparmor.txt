  signal (send) set=(kill,term,int,hup,cont),

  /usr/local/bin/** ix,

  /tmp/** rwk,

  /usr/lib/python3*/** r,

  # Node.js (Claude Agent SDK)
  /usr/lib/node_modules/** rix,

  # nginx

  /var/log/nginx/** rw,
  /var/lib/nginx/** rw,

  network netlink raw,

  # Proc and sys
  /proc/** r,
